How do get nth Fibonacci number?

Psudocode

  1. Make Array
  2. Loop from bottom up and get current with -2 pointer, -1pointer


DEBUG: 
function fibonacci(n){
  var fibo = [0, 1];
  
  if (n <= 2) return 1;

  for (var i = 2; i <=n; i++ ){
   fibo[i] = fibo[i-1]+fibo[i-2];
  }

 return fibo[n];
} 

//RECURSIVE
function fibonacciR(n){
  if(n<=1)
    return n;
  else
    return fibonacciR(n-1) + fibonacciR (n-2);  
}
